        const sendLink = document.getElementById('sendLink');

        const myEmail = "meher.radio.tun@gmail.com"; [cite: 6]

        function updateMailto() {
            const subject = encodeURIComponent(subjectInput.value || "Professional Inquiry");
            const body = encodeURIComponent(messageInput.value || "");
            sendLink.href = `mailto:${myEmail}?subject=${subject}&body=${body}`;
        }

        // Initialisation et écoute des changements
        updateMailto();
        subjectInput.addEventListener('input', updateMailto);
        messageInput.addEventListener('input', updateMailto);

        // Sécurité pour éviter l'envoi à vide
        sendLink.addEventListener('click', (e) => {
            if (!subjectInput.value || !messageInput.value) {
                e.preventDefault();
                alert("Please fill in both the subject and the message.");
            }
